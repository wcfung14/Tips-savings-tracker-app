📋 Step 1 — Set up your Apps Script
Do this once in Google Sheets. Takes about 3 minutes.
  1. Open Google Sheets, create a blank spreadsheet
  2. Add headers in Row 1: id date restaurant food amount tipped pct timestamp
  3. Click Extensions → Apps Script, delete everything, paste the code below
  4. Save, then Deploy → New deployment → Web app
  5. Set Execute as: Me · Who has access: Anyone → Deploy
  6. Copy the Web app URL and paste it below ↓
const SHEET_NAME = 'Sheet1'; function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const data = rows.slice(1).map(row => { const o = {}; headers.forEach((h,i) => o[h]=row[i]); return o; }); return ContentService.createTextOutput(JSON.stringify({success:true,data})).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const payload = JSON.parse(e.postData.contents); if (payload.action === 'save') { sheet.clearContents(); sheet.appendRow(['id','date','restaurant','food','amount','tipped','pct','timestamp']); payload.entries.forEach(entry => sheet.appendRow([entry.id,entry.date,entry.restaurant,entry.food,entry.amount,entry.tipped,entry.pct,new Date().toISOString()])); return ContentService.createTextOutput(JSON.stringify({success:true})).setMimeType(ContentService.MimeType.JSON); } }
